        \frametitle{Principe de la marche aléatoire}
        \center
        On utilise un générateur aléatoire pour engendrer $\alpha$.
        \begin{minipage}{0.5\textwidth}
            \SetAlFnt{\tiny}
            \SetAlCapFnt{\tiny}
            \SetAlCapNameFnt{\tiny}
            \begin{algorithm}[H]
                \tiny
                \caption{\tiny Simulation de mouvement aléatoire} 

                \SetKwData{Temps}{temps}
                \SetKwData{Tf}{temps\_final}
                \SetKwData{Dt}{dt}
                \SetKwData{V}{v}
                \SetKwData{Ens}{T}
                \SetKwInOut{Input}{Entrée}

                \Input{un ensemble de points \Ens, vitesse \V, pas de temps \Dt, \Tf}
                \BlankLine

                \Temps $\leftarrow$ 0\;
                \While{\Temps $<$ \Tf}{
                    \ForEach{point $P$ dans \Ens}{
                        Choisir $\alpha \in [0, 2\pi]$\;
                        Déplacer $P$ de la distance \V dans la direction $\alpha$\;
                    }
                    \Temps $\leftarrow$ \Temps + \Dt\;
                }
                \end{algorithm}
        \end{minipage}

    \subsection{Définition}
    \begin{frame}
        \frametitle{Définition}
        \begin{center}
            Soit $a, b, q \in \mathbb{N}$, \, 
                $
                \begin{cases}
                u_0 \in \mathbb{N}\\
                u_{n+1} = (a \cdot u_n + b) \bmod q 
                \end{cases}
                $
            \vspace{1cm}
            \\
            Exemples :
            \begin{itemize}
                \item Marsaglia : $a = \text{69 069,} \ b = 0, \ q = 2^{32}$
                \item Knuth et Lewis : $a = \text{1 664 525,} \ b = \text{1 013 904 223,} \ q = 2^{32}$
            \end{itemize}
        \end{center}
    \end{frame}

    \subsection{Test de la distance entre deux points d'un carré}
    \begin{frame}
        \frametitle{Principe}
        \begin{minipage}{0.5\textwidth}
            \SetAlFnt{\tiny}
            \SetAlCapFnt{\tiny}
            \SetAlCapNameFnt{\tiny}
            \begin{algorithm}[H]
                \tiny
                \caption{\tiny Test de la densité dans un carré} 

                \SetKwData{Lnba}{liste de nombres aléatoires}
                \SetKwData{n}{n}
                \SetKwInOut{Input}{Entrée}
                \SetKwInOut{Output}{Sortie}

                \Input{\Lnba, un entier \n}
                \BlankLine

                L $\leftarrow$ \n points choisit aléatoirement dans un carré\\
                L\_dits $\leftarrow$ []\\
                \ForEach{point $P$ dans L}{
                    dist $\leftarrow$ toute les distances entre P et les autres points;\\
                    L\_dist $\leftarrow$ L\_dist + dist;
                }
                nb\_inter $\leftarrow$ compte le nombre de valeur de L\_dist par subdivision;\\
                f\_répartition $\leftarrow$ approche la fonction de répartition avec nb\_inter;\\
                f\_densité $\leftarrow$ approche la fonction de densité avec f\_répartition;\\
                \Output{distance entre f\_densité et la densité théorique}
                \end{algorithm}
